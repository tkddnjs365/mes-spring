#### 매번 서버에 jar 파일 배포 ####


1. 애플리케이션 빌드
  - ./gradlew clean bootJar

2. 서버에 해당 파일 이동

#### 서버에 배포 ####
# 서버에서 실행 중인 Java 프로세스 종료
Stop-Process -Name "java" -Force

# 애플리케이션 빌드
./gradlew clean bootJar

# 백그라운드 실행
cd D:\mes_boot
Start-Process java -ArgumentList "-jar","mes_boot-0.0.1-SNAPSHOT.jar" -WindowStyle Hidden -RedirectStandardOutput "app.log"

# 2. 실행 확인 (몇 초 후)
Get-Process -Name "java"
netstat -an | Select-String ":7010"

# 3. Next.js에서 테스트
# http://서버IP:7010/your-api-endpoint

# 4. 로그 확인
Get-Content -Path "D:\mes_boot\app.log" -Tail 10 -Wait
